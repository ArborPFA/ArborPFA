132  return pandora::STATUS_CODE_NOT_FOUND;
133

271  const float barrelOuterZMuon = std::fabs(pGeometryManager->GetSubDetector(pandora::MUON_BARREL).GetOuterZCoordinate());

273 
274  const bool IsEnclosingEndCap(endCapOuterR > m_barrelInnerR); // true for ILD !
275  m_rCorrection = ((!IsEnclosingEndCap) ? 0.f : m_barrelInnerR * ((m_endCapInnerZ / barrelOuterZ) - 1.f));
276  m_zCorrection = ((IsEnclosingEndCap) ? 0.f : m_endCapInnerZ * ((m_barrelInnerR / endCapOuterR) - 1.f));
277  m_rCorrectionMuon = ((!IsEnclosingEndCap) ? 0.f : m_barrelInnerRMuon * ((m_endCapInnerZMuon / barrelOuterZMuon) - 1.f));
278  m_zCorrectionMuon = ((IsEnclosingEndCap) ? 0.f : m_endCapInnerZMuon * ((m_barrelInnerRMuon / endCapOuterRMuon) - 1.f));

284  {

302  void ArborPseudoLayerPlugin::FillAngleVector(const unsigned int symmetryOrder, const float phi0, AngleVector &angleVector) const
303  {
304  const float twoPi = static_cast<float>(2. * std::acos(-1.));
