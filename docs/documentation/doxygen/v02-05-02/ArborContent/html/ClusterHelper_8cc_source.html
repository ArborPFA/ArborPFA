567  pandora::StatusCode ClusterHelper::GetDistanceToDetectorGap(const pandora::CartesianVector &positionVector, const pandora::DetectorGap *const pDetectorGap, float &distanceToGap)
568  {
569  distanceToGap = 0.f;
570 
571  if(NULL == pDetectorGap)
572  return pandora::STATUS_CODE_INVALID_PARAMETER;
573 
574  if(pDetectorGap->IsInGap(positionVector, pandora::ECAL))
575  return pandora::STATUS_CODE_FAILURE;
576 
577  const pandora::BoxGap *const pBoxGap(dynamic_cast<const pandora::BoxGap *const>(pDetectorGap));
578 
579  if(pBoxGap)
580  {
581  const pandora::CartesianVector relativePosition(positionVector - pBoxGap->GetVertex());
582 
583  const float projection1(relativePosition.GetDotProduct(pBoxGap->GetSide1().GetUnitVector()));
584  const float projection2(relativePosition.GetDotProduct(pBoxGap->GetSide2().GetUnitVector()));
585  const float projection3(relativePosition.GetDotProduct(pBoxGap->GetSide3().GetUnitVector()));
586 
587  distanceToGap = std::min(fabs(projection1), std::min(fabs(projection2), fabs(projection3)));
588 
589  return pandora::STATUS_CODE_SUCCESS;
590  }
591 
592  const pandora::ConcentricGap *const pConcentricGap(dynamic_cast<const pandora::ConcentricGap *const>(pDetectorGap));
593 
594  if(pConcentricGap)
595  {
596  const float z(positionVector.GetZ());
597  const float x(positionVector.GetX()), y(positionVector.GetY());
598  const float r(std::sqrt(x * x + y * y));
599  bool inner(false), outer(false);
600  const bool inZ(pConcentricGap->GetMaxZCoordinate() < z && pConcentricGap->GetMinZCoordinate() > z);
601  float zDistance(0.f), rInnerDistance(0.f), rOuterDistance(0.f);
602 
603  if(pConcentricGap->GetMaxZCoordinate() < z)
604  {
605  zDistance = fabs(z - pConcentricGap->GetMaxZCoordinate());
606  }
607 
608  if(pConcentricGap->GetMinZCoordinate() > z)
609  {
610  zDistance = fabs(z - pConcentricGap->GetMinZCoordinate());
611  }
612 
613  // inner r case
614  for(unsigned int i=0 ; i<pConcentricGap->GetInnerSymmetryOrder()+1 ; i++)
615  {
616  const float phi = pConcentricGap->GetInnerPhiCoordinate() + 2 * M_PI * (static_cast<float>(i) / static_cast<float>(pConcentricGap->GetInnerSymmetryOrder()));
617  const float phiMin = phi - (2 * M_PI / static_cast<float>(pConcentricGap->GetInnerSymmetryOrder())) / 2.f;
618  const float phiMax = phi + (2 * M_PI / static_cast<float>(pConcentricGap->GetInnerSymmetryOrder())) / 2.f;
619  const float phiPosition(std::atan2(y, x));
620 
621  if(phiPosition > phiMin && phiPosition < phiMax)
622  {
623  const pandora::CartesianVector normaleVector(std::cos(phi), std::sin(phi), 0.f);
624 
625  if(r*std::cos(normaleVector.GetOpeningAngle(positionVector)) < pConcentricGap->GetInnerRCoordinate())
626  {
627  rInnerDistance = fabs(pConcentricGap->GetInnerRCoordinate() - r*std::cos(normaleVector.GetOpeningAngle(positionVector)));
628  inner = true;
629  }
630 
631  break;
632  }
633  }
634 
635  // outer r case
636  for(unsigned int i=0 ; i<pConcentricGap->GetOuterSymmetryOrder()+1 ; i++)
637  {
638  const float phi = pConcentricGap->GetOuterPhiCoordinate() + 2 * M_PI * (static_cast<float>(i) / static_cast<float>(pConcentricGap->GetOuterSymmetryOrder()));
639  const float phiMin = phi - (2 * M_PI / static_cast<float>(pConcentricGap->GetOuterSymmetryOrder())) / 2.f;
640  const float phiMax = phi + (2 * M_PI / static_cast<float>(pConcentricGap->GetOuterSymmetryOrder())) / 2.f;
641  const float phiPosition(std::atan2(y, x));
642 
643  if(phiPosition > phiMin && phiPosition < phiMax)
644  {
645  const pandora::CartesianVector normaleVector(std::cos(phi), std::sin(phi), 0.f);
646 
647  if(r*std::cos(normaleVector.GetOpeningAngle(positionVector)) > pConcentricGap->GetOuterRCoordinate())
648  {
649  rOuterDistance = fabs(r*std::cos(normaleVector.GetOpeningAngle(positionVector)) - pConcentricGap->GetOuterRCoordinate());
650  outer = true;
651  }
652 
653  break;
654  }
655  }
656 
657  if(!inner && !outer)
658  {
659  distanceToGap = fabs(zDistance);
660  }
661  else if(inZ)
662  {
663  if(inner)
664  distanceToGap = fabs(rInnerDistance);
665  else
666  distanceToGap = fabs(rOuterDistance);
667  }
668  else
669  {
670  float rDistance(inner ? rInnerDistance : rOuterDistance);
671  distanceToGap = std::sqrt(fabs(rDistance)*fabs(rDistance) + fabs(zDistance)*fabs(zDistance));
672  }
673 
674  return pandora::STATUS_CODE_SUCCESS;
675  }
676 
677  return pandora::STATUS_CODE_FAILURE;
678  }
