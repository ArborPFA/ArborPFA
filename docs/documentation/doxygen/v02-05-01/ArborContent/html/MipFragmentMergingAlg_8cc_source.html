268  pandora::StatusCode MipFragmentMergingAlg::FindChargedMipFragments(const pandora::ClusterVector &clusterVector, ClusterToClusterMap &clusterToClusterMap)
269  {
270  pandora::ClusterList associatedDaughterClusterList;
271 
272  for(pandora::ClusterVector::const_iterator iIter = clusterVector.begin(), iEndIter = clusterVector.end() ;
273  iEndIter != iIter ; ++iIter)
274  {
275  const pandora::Cluster *const pParentCluster(*iIter);
276 
277  // mip fragment ?
278  if(!this->IsMipFragment(pParentCluster))
279  continue;
280 
281  if(pParentCluster->GetAssociatedTrackList().empty())
282  continue;
283 
284  const unsigned int parentOuterPseudoLayer(pParentCluster->GetOuterPseudoLayer());
285 
286  pandora::ClusterFitResult parentClusterFitResult;
287 
288  if(pandora::STATUS_CODE_SUCCESS != pandora::ClusterFitHelper::FitEnd(pParentCluster, m_nParentFitPseudoLayers, parentClusterFitResult))
289  continue;
290 
291  if(!parentClusterFitResult.IsFitSuccessful())
292  continue;
293 
294  const pandora::CartesianVector &parentClusterDirection(parentClusterFitResult.GetDirection());
295  const pandora::CartesianVector parentEndPoint(pParentCluster->GetCentroid(parentOuterPseudoLayer));
296 
297  for(pandora::ClusterVector::const_iterator jIter = clusterVector.begin(), jEndIter = clusterVector.end() ;
298  jEndIter != jIter ; ++jIter)
299  {
300  const pandora::Cluster *const pDaughterCluster(*jIter);
301 
302  if(pDaughterCluster == pParentCluster)
303  continue;
304 
305  const unsigned int daughterInnerPseudoLayer(pDaughterCluster->GetInnerPseudoLayer());
306 
307  if(daughterInnerPseudoLayer <= parentOuterPseudoLayer)
308  continue;
309 
310  if(associatedDaughterClusterList.find(pDaughterCluster) != associatedDaughterClusterList.end())
311  continue;
312 
313  float oldChi(0.f), newChi(0.f);
314 
315  if(pandora::STATUS_CODE_SUCCESS != ClusterHelper::GetChiClusterMerging(this->GetPandora(), pParentCluster, pDaughterCluster, oldChi, newChi))
316  continue;
317 
318  // check for track cluster E-p compatibility improvement
319  if(oldChi*oldChi < newChi*newChi)
320  continue;
321 
322  const pandora::CartesianVector daughterStartPoint(pDaughterCluster->GetCentroid(daughterInnerPseudoLayer));
323  const float cosineDirection( (daughterStartPoint-parentEndPoint).GetCosOpeningAngle(parentClusterDirection) );
324  const unsigned int pseudoLayerDifference(daughterInnerPseudoLayer-parentOuterPseudoLayer);
325  bool possibleAssociation(false);
326 
327  if((pseudoLayerDifference < m_maxClusterSeparationPseudoLayer) && (cosineDirection > m_maxCosineDirection))
328  {
329  possibleAssociation = true;
330  }
331  else if((pseudoLayerDifference < m_maxClusterSeparationPseudoLayer2) && (cosineDirection > m_maxCosineDirection2))
332  {
333  possibleAssociation = true;
334  }
335  else if((pseudoLayerDifference < m_maxClusterSeparationPseudoLayer3) && (cosineDirection > m_maxCosineDirection3))
336  {
337  possibleAssociation = true;
338  }
339 
340  if(possibleAssociation)
341  {
342  if(!associatedDaughterClusterList.insert(pDaughterCluster).second)
343  continue;
344 
345  clusterToClusterMap[pDaughterCluster] = pParentCluster;
346  }
347  }
348  }
349 
350  return pandora::STATUS_CODE_SUCCESS;
351  }
