147  void PerfectParticleFlowAlgorithm::FullCaloHitCollection(const MCParticle *const pPfoTarget, const CaloHit *const pCaloHit, CaloHitList &caloHitList) const
148  {
149  const MCParticleWeightMap mcParticleWeightMap(pCaloHit->GetMCParticleWeightMap());
150 
151  if (mcParticleWeightMap.size() < 2)
152  return this->SimpleCaloHitCollection(pPfoTarget, pCaloHit, caloHitList);
153 
154  float mcParticleWeightSum(0.f);
155 
156  for (MCParticleWeightMap::const_iterator wtIter = mcParticleWeightMap.begin(), wtIterEnd = mcParticleWeightMap.end(); wtIter != wtIterEnd; ++wtIter)
157  mcParticleWeightSum += wtIter->second;
158 
159  if (mcParticleWeightSum < std::numeric_limits<float>::epsilon())
160  throw StatusCodeException(STATUS_CODE_FAILURE);
161 
162  const CaloHit *pLocalCaloHit = pCaloHit;
163 
164  for (MCParticleWeightMap::const_iterator wtIter = mcParticleWeightMap.begin(), wtIterEnd = mcParticleWeightMap.end(); wtIter != wtIterEnd; ++wtIter)
165  {
166  const MCParticle *const pHitMCParticle(wtIter->first);
167  const MCParticle *const pHitPfoTarget(pHitMCParticle->GetPfoTarget());
168  const float weight(wtIter->second);
169 
170  if (pHitPfoTarget != pPfoTarget)
171  continue;
172 
173  const CaloHit *pCaloHitToAdd = pLocalCaloHit;
174 
175  if (pCaloHitToAdd->GetWeight() < std::numeric_limits<float>::epsilon())
176  throw StatusCodeException(STATUS_CODE_FAILURE);
177 
178  const float weightFraction(weight / (mcParticleWeightSum * pCaloHitToAdd->GetWeight()));
179  const bool isBelowThreshold((weightFraction - m_minWeightFraction) < std::numeric_limits<float>::epsilon());
180 
181  if (isBelowThreshold)
182  continue;
183 
184  const bool shouldFragment(weightFraction + m_minWeightFraction - 1.f < std::numeric_limits<float>::epsilon());
185 
186  if (shouldFragment)
187  PANDORA_THROW_RESULT_IF(STATUS_CODE_SUCCESS, !=, PandoraContentApi::Fragment(*this, pLocalCaloHit, weightFraction, pCaloHitToAdd, pLocalCaloHit));
188 
189  caloHitList.insert(pCaloHitToAdd);
190  }

248  const float electronMass(PdgTable::GetParticleMass(E_MINUS));
249 
250  charge += pTrack->GetCharge();
251  momentum += pTrack->GetMomentumAtDca();
252  energyWithPionMass += pTrack->GetEnergyAtDca();

262  pfoParameters.m_energy = (pfoParameters.m_particleId.Get() == PHOTON) ? energyWithElectronMass : energyWithPionMass;
263  pfoParameters.m_mass = std::sqrt(std::max(pfoParameters.m_energy.Get() * pfoParameters.m_energy.Get() - pfoParameters.m_momentum.Get().GetDotProduct(pfoParameters.m_momentum.Get()), 0.f));

307  const CartesianVector PerfectParticleFlowAlgorithm::GetEnergyWeightedCentroid(const Cluster *const pCluster, const unsigned int innerPseudoLayer,
308  const unsigned int outerPseudoLayer) const
309  {
310  float energySum(0.f);
311  CartesianVector energyPositionSum(0.f, 0.f, 0.f);
312  const OrderedCaloHitList &orderedCaloHitList(pCluster->GetOrderedCaloHitList());
313 
314  for (OrderedCaloHitList::const_iterator iter = orderedCaloHitList.begin(), iterEnd = orderedCaloHitList.end(); iter != iterEnd; ++iter)
315  {
316  if (iter->first > outerPseudoLayer)
317  break;
318 
319  if (iter->first < innerPseudoLayer)
320  continue;
321 
322  for (CaloHitList::const_iterator hitIter = iter->second->begin(), hitIterEnd = iter->second->end(); hitIter != hitIterEnd; ++hitIter)
323  {
324  const float electromagneticEnergy((*hitIter)->GetElectromagneticEnergy());
325  energySum += electromagneticEnergy;
326  energyPositionSum += ((*hitIter)->GetPositionVector() * electromagneticEnergy);
327  }
328  }
329 
330  if (energySum < std::numeric_limits<float>::epsilon())
331  throw StatusCodeException(STATUS_CODE_NOT_INITIALIZED);
332 
333  return (energyPositionSum * (1.f / energySum));
334  }
