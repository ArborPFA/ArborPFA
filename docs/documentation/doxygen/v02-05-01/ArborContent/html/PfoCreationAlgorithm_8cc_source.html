159 pandora::StatusCode PfoCreationAlgorithm::SetSiblingTrackBasedPfoParameters(const pandora::Track *const pTrack, PfoParameters &pfoParameters) const
160 {
161  int charge(0);
162  float energy(0.f);
163  pandora::CartesianVector momentum(0.f, 0.f, 0.f);
164 
165  pandora::TrackList fullSiblingTrackList(pTrack->GetSiblingTrackList());
166  fullSiblingTrackList.insert(pTrack);
167 
168  for (pandora::TrackList::const_iterator iter = fullSiblingTrackList.begin(), iterEnd = fullSiblingTrackList.end(); iter != iterEnd; ++iter)
169  {
170  const pandora::Track *const pSiblingTrack = *iter;
171  charge += pSiblingTrack->GetCharge();
172 
173  if (!pSiblingTrack->CanFormPfo() && !pSiblingTrack->CanFormClusterlessPfo())
174  continue;
175 
176  // ATTN Assume sibling-track-based pfos represent pair-production
177  const float electronMass(pandora::PdgTable::GetParticleMass(pandora::E_MINUS));
178  energy += std::sqrt(electronMass * electronMass + pSiblingTrack->GetMomentumAtDca().GetMagnitudeSquared());
179  momentum += pSiblingTrack->GetMomentumAtDca();
180  }
181 
182  if (energy < std::numeric_limits<float>::epsilon())
183  return pandora::STATUS_CODE_NOT_INITIALIZED;
184 
185  pfoParameters.m_energy = energy;
186  pfoParameters.m_momentum = momentum;
187  pfoParameters.m_mass = std::sqrt(std::max(energy * energy - momentum.GetDotProduct(momentum), 0.f));
188  pfoParameters.m_charge = charge;
189  pfoParameters.m_particleId = pandora::PHOTON;
190 
191  return pandora::STATUS_CODE_SUCCESS;

241 pandora::StatusCode PfoCreationAlgorithm::CreateNeutralPfos() const
242 {
243  const pandora::ClusterList *pClusterList = NULL;
244  PANDORA_RETURN_RESULT_IF(pandora::STATUS_CODE_SUCCESS, !=, PandoraContentApi::GetCurrentList(*this, pClusterList));
245 
246  // Examine clusters with no associated tracks to form neutral pfos
247  for (pandora::ClusterList::const_iterator iter = pClusterList->begin(), iterEnd = pClusterList->end(); iter != iterEnd; ++iter)
248  {
249  const pandora::Cluster *const pCluster = *iter;
250 
251  if (!pCluster->GetAssociatedTrackList().empty())
252  continue;
253 
254  if (pCluster->GetNCaloHits() < m_minHitsInCluster)
255  continue;
256 
257  const bool isPhoton(pCluster->IsPhotonFast(this->GetPandora()));
258  float clusterEnergy(isPhoton ? pCluster->GetCorrectedElectromagneticEnergy(this->GetPandora()) : pCluster->GetCorrectedHadronicEnergy(this->GetPandora()));
259 
260  // Veto non-photon clusters below hadronic energy threshold and those occupying a single layer
261  if (!isPhoton)
262  {
263  if (clusterEnergy < m_minClusterHadronicEnergy)
264  continue;
265 
266  if (!m_allowSingleLayerClusters && (pCluster->GetInnerPseudoLayer() == pCluster->GetOuterPseudoLayer()))
267  continue;
268  }
269  else
270  {
271  if (clusterEnergy < m_minClusterElectromagneticEnergy)
272  continue;
273  }
274 
275  // Specify the pfo parameters
276  PandoraContentApi::ParticleFlowObject::Parameters pfoParameters;
277  pfoParameters.m_particleId = (isPhoton ? pandora::PHOTON : pandora::NEUTRON);
278  pfoParameters.m_charge = 0;
279  pfoParameters.m_mass = (isPhoton ? pandora::PdgTable::GetParticleMass(pandora::PHOTON) : pandora::PdgTable::GetParticleMass(pandora::NEUTRON));
280  pfoParameters.m_energy = clusterEnergy;
281  pfoParameters.m_clusterList.insert(pCluster);
282 
283  // Photon position: 0) unweighted inner centroid, 1) energy-weighted inner centroid, 2+) energy-weighted centroid for all layers
284  pandora::CartesianVector positionVector(0.f, 0.f, 0.f);
285  const unsigned int clusterInnerLayer(pCluster->GetInnerPseudoLayer());
286 
287  if (!isPhoton || (0 == m_photonPositionAlgorithm))
288  {
289  positionVector = pCluster->GetCentroid(clusterInnerLayer);
290  }
291  else if (1 == m_photonPositionAlgorithm)
292  {
293  positionVector = this->GetEnergyWeightedCentroid(pCluster, clusterInnerLayer, clusterInnerLayer);
294  }
295  else
296  {
297  positionVector = this->GetEnergyWeightedCentroid(pCluster, clusterInnerLayer, pCluster->GetOuterPseudoLayer());
298  }
299 
300  const pandora::CartesianVector momentum(positionVector.GetUnitVector() * clusterEnergy);
301  pfoParameters.m_momentum = momentum;
302 
303  const pandora::ParticleFlowObject *pPfo(NULL);
304  PANDORA_RETURN_RESULT_IF(pandora::STATUS_CODE_SUCCESS, !=, PandoraContentApi::ParticleFlowObject::Create(*this, pfoParameters, pPfo));
305  }
306 
307  return pandora::STATUS_CODE_SUCCESS;
308 }

312 const pandora::CartesianVector PfoCreationAlgorithm::GetEnergyWeightedCentroid(const pandora::Cluster *const pCluster, const unsigned int innerPseudoLayer,
313  const unsigned int outerPseudoLayer) const
314 {
315  float energySum(0.f);
316  pandora::CartesianVector energyPositionSum(0.f, 0.f, 0.f);
317  const pandora::OrderedCaloHitList &orderedCaloHitList(pCluster->GetOrderedCaloHitList());
318 
319  for (pandora::OrderedCaloHitList::const_iterator iter = orderedCaloHitList.begin(), iterEnd = orderedCaloHitList.end(); iter != iterEnd; ++iter)
320  {
321  if (iter->first > outerPseudoLayer)
322  break;
323 
324  if (iter->first < innerPseudoLayer)
325  continue;
326 
327  for (pandora::CaloHitList::const_iterator hitIter = iter->second->begin(), hitIterEnd = iter->second->end(); hitIter != hitIterEnd; ++hitIter)
328  {
329  const float electromagneticEnergy((*hitIter)->GetElectromagneticEnergy());
330  energySum += electromagneticEnergy;
331  energyPositionSum += ((*hitIter)->GetPositionVector() * electromagneticEnergy);
332  }
333  }
334 
335  if (energySum < std::numeric_limits<float>::epsilon())
336  throw pandora::StatusCodeException(pandora::STATUS_CODE_NOT_INITIALIZED);
337 
338  return (energyPositionSum * (1.f / energySum));
339 }
